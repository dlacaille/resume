% Columns

% Date calculations
\newcounter{dateone}
\newcounter{datetwo}
\newcommand{\yearssince}[3]{%
  \setmydatenumber{dateone}{\the\year}{\the\month}{\the\day}%
  \setmydatenumber{datetwo}{#1}{#2}{#3}%
  \addtocounter{datetwo}{-\thedateone}%
  \the\numexpr-\thedatetwo/365\relax\space%
}

}

  \uppercase{#1}\par
